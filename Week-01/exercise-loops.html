JavaScript Loop Examples

1. for loop — Print numbers 1 to 5

for (let i = 1; i <= 5; i++) {
  console.log(i);
}
1
2
3
4
5

2. while loop — Count down from 5 to 1

let i = 5;
while (i > 0) {
  console.log(i);
  i--;
}
5
4
3
2
1

3. do...while loop — Run at least once

let i = 0;
do {
  console.log("Run:", i);
  i++;
} while (i < 3);
Run: 0
Run: 1
Run: 2

4. forEach — Print array elements

const fruits = ["apple", "banana", "cherry"];
fruits.forEach(function(fruit) {
  console.log(fruit);
});
apple
banana
cherry

5. for...in — Loop through object keys

const person = { name: "Liam", age: 28, city: "Calgary" };
for (let key in person) {
  console.log(key + ": " + person[key]);
}
name: Liam
age: 28
city: Calgary

6. for...of — Loop through values of an array

const colors = ["red", "green", "blue"];
for (let color of colors) {
  console.log(color);
}
red
green
blue
